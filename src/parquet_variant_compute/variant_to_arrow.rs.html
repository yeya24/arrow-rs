32/// Builder for converting variant values to primitive Arrow arrays. It is used by both

57/// Builder for converting variant values into strongly typed Arrow arrays.
58///
59/// Useful for variant_get kernels that need to extract specific paths from variant values, possibly
60/// with casting of leaf values to specific types.
61pub(crate) enum VariantToArrowRowBuilder<'a> {
62    Primitive(PrimitiveVariantToArrowRowBuilder<'a>),
63    BinaryVariant(VariantToBinaryVariantArrowRowBuilder),
64
65    // Path extraction wrapper - contains a boxed enum for any of the above
66    WithPath(VariantPathRowBuilder<'a>),
67}

169/// Creates a primitive row builder, returning Err if the requested data type is not primitive.
170pub(crate) fn make_primitive_variant_to_arrow_row_builder<'a>(
171    data_type: &'a DataType,
172    cast_options: &'a CastOptions,
173    capacity: usize,
174) -> Result<PrimitiveVariantToArrowRowBuilder<'a>> {
175    use PrimitiveVariantToArrowRowBuilder::*;
176
177    let builder = match data_type {
178        DataType::Boolean => Boolean(VariantToBooleanArrowRowBuilder::new(cast_options, capacity)),
179        DataType::Int8 => Int8(VariantToPrimitiveArrowRowBuilder::new(
180            cast_options,
181            capacity,
182        )),
183        DataType::Int16 => Int16(VariantToPrimitiveArrowRowBuilder::new(
184            cast_options,
185            capacity,
186        )),
187        DataType::Int32 => Int32(VariantToPrimitiveArrowRowBuilder::new(
188            cast_options,
189            capacity,
190        )),
191        DataType::Int64 => Int64(VariantToPrimitiveArrowRowBuilder::new(
192            cast_options,
193            capacity,
194        )),
195        DataType::UInt8 => UInt8(VariantToPrimitiveArrowRowBuilder::new(
196            cast_options,
197            capacity,
198        )),
199        DataType::UInt16 => UInt16(VariantToPrimitiveArrowRowBuilder::new(
200            cast_options,
201            capacity,
202        )),
203        DataType::UInt32 => UInt32(VariantToPrimitiveArrowRowBuilder::new(
204            cast_options,
205            capacity,
206        )),
207        DataType::UInt64 => UInt64(VariantToPrimitiveArrowRowBuilder::new(
208            cast_options,
209            capacity,
210        )),
211        DataType::Float16 => Float16(VariantToPrimitiveArrowRowBuilder::new(
212            cast_options,
213            capacity,
214        )),
215        DataType::Float32 => Float32(VariantToPrimitiveArrowRowBuilder::new(
216            cast_options,
217            capacity,
218        )),
219        DataType::Float64 => Float64(VariantToPrimitiveArrowRowBuilder::new(
220            cast_options,
221            capacity,
222        )),
223        DataType::Timestamp(TimeUnit::Microsecond, None) => TimestampMicroNtz(
224            VariantToTimestampNtzArrowRowBuilder::new(cast_options, capacity),
225        ),
226        DataType::Timestamp(TimeUnit::Microsecond, tz) => TimestampMicro(
227            VariantToTimestampArrowRowBuilder::new(cast_options, capacity, tz.clone()),
228        ),
229        DataType::Timestamp(TimeUnit::Nanosecond, None) => TimestampNanoNtz(
230            VariantToTimestampNtzArrowRowBuilder::new(cast_options, capacity),
231        ),
232        DataType::Timestamp(TimeUnit::Nanosecond, tz) => TimestampNano(
233            VariantToTimestampArrowRowBuilder::new(cast_options, capacity, tz.clone()),
234        ),
235        DataType::Date32 => Date(VariantToPrimitiveArrowRowBuilder::new(
236            cast_options,
237            capacity,
238        )),
239        _ if data_type.is_primitive() => {
240            return Err(ArrowError::NotYetImplemented(format!(
241                "Primitive data_type {data_type:?} not yet implemented"
242            )));
243        }
244        _ => {
245            return Err(ArrowError::InvalidArgumentError(format!(
246                "Not a primitive type: {data_type:?}"
247            )));
248        }
249    };
250    Ok(builder)
251}

253pub(crate) fn make_variant_to_arrow_row_builder<'a>(
254    metadata: &BinaryViewArray,
255    path: VariantPath<'a>,
256    data_type: Option<&'a DataType>,
257    cast_options: &'a CastOptions,
258    capacity: usize,
259) -> Result<VariantToArrowRowBuilder<'a>> {
260    use VariantToArrowRowBuilder::*;
261
262    let mut builder = match data_type {
263        // If no data type was requested, build an unshredded VariantArray.
264        None => BinaryVariant(VariantToBinaryVariantArrowRowBuilder::new(
265            metadata.clone(),
266            capacity,
267        )),
268        Some(DataType::Struct(_)) => {
269            return Err(ArrowError::NotYetImplemented(
270                "Converting unshredded variant objects to arrow structs".to_string(),
271            ));
272        }
273        Some(
274            DataType::List(_)
275            | DataType::LargeList(_)
276            | DataType::ListView(_)
277            | DataType::LargeListView(_)
278            | DataType::FixedSizeList(..),
279        ) => {
280            return Err(ArrowError::NotYetImplemented(
281                "Converting unshredded variant arrays to arrow lists".to_string(),
282            ));
283        }
284        Some(data_type) => {
285            let builder =
286                make_primitive_variant_to_arrow_row_builder(data_type, cast_options, capacity)?;
287            Primitive(builder)
288        }
289    };
290
291    // Wrap with path extraction if needed
292    if !path.is_empty() {
293        builder = WithPath(VariantPathRowBuilder {
294            builder: Box::new(builder),
295            path,
296        })
297    };
298
299    Ok(builder)
300}

302/// A thin wrapper whose only job is to extract a specific path from a variant value and pass the
303/// result to a nested builder.
304pub(crate) struct VariantPathRowBuilder<'a> {
305    builder: Box<VariantToArrowRowBuilder<'a>>,
306    path: VariantPath<'a>,
307}
308
309impl<'a> VariantPathRowBuilder<'a> {
310    fn append_null(&mut self) -> Result<()> {
311        self.builder.append_null()
312    }
313
314    fn append_value(&mut self, value: Variant<'_, '_>) -> Result<bool> {
315        if let Some(v) = value.get_path(&self.path) {
316            self.builder.append_value(v)
317        } else {
318            self.builder.append_null()?;
319            Ok(false)
320        }
321    }
322
323    fn finish(self) -> Result<ArrayRef> {
324        self.builder.finish()
325    }
326}

328/// Helper function to get a user-friendly type name
329fn get_type_name<T: ArrowPrimitiveType>() -> &'static str {
330    match std::any::type_name::<T>() {
331        "arrow_array::types::Int32Type" => "Int32",
332        "arrow_array::types::Int16Type" => "Int16",
333        "arrow_array::types::Int8Type" => "Int8",
334        "arrow_array::types::Int64Type" => "Int64",
335        "arrow_array::types::UInt32Type" => "UInt32",
336        "arrow_array::types::UInt16Type" => "UInt16",
337        "arrow_array::types::UInt8Type" => "UInt8",
338        "arrow_array::types::UInt64Type" => "UInt64",
339        "arrow_array::types::Float32Type" => "Float32",
340        "arrow_array::types::Float64Type" => "Float64",
341        "arrow_array::types::Float16Type" => "Float16",
342        "arrow_array::types::TimestampMicrosecondType" => "Timestamp(Microsecond)",
343        "arrow_array::types::TimestampNanosecondType" => "Timestamp(Nanosecond)",
344        "arrow_array::types::Date32Type" => "Date32",
345        _ => "Unknown",
346    }
347}

349macro_rules! define_variant_to_primitive_builder {
350    (struct $name:ident<$lifetime:lifetime $(, $generic:ident: $bound:path )?>
351    |$array_param:ident $(, $field:ident: $field_type:ty)?| -> $builder_name:ident $(< $array_type:ty >)? { $init_expr: expr },
352    |$value: ident| $value_transform:expr,
353    type_name: $type_name:expr) => {
354        pub(crate) struct $name<$lifetime $(, $generic : $bound )?>
355        {
356            builder: $builder_name $(<$array_type>)?,
357            cast_options: &$lifetime CastOptions<$lifetime>,
358        }
359
360        impl<$lifetime $(, $generic: $bound+ )?> $name<$lifetime $(, $generic )?> {
361            fn new(
362                cast_options: &$lifetime CastOptions<$lifetime>,
363                $array_param: usize,
364                // add this so that $init_expr can use it
365                $( $field: $field_type, )?
366            ) -> Self {
367                Self {
368                    builder: $init_expr,
369                    cast_options,
370                }
371            }
372
373            fn append_null(&mut self) -> Result<()> {
374                self.builder.append_null();
375                Ok(())
376            }
377
378            fn append_value(&mut self, $value: &Variant<'_, '_>) -> Result<bool> {
379                if let Some(v) = $value_transform {
380                    self.builder.append_value(v);
381                    Ok(true)
382                } else {
383                    if !self.cast_options.safe {
384                        // Unsafe casting: return error on conversion failure
385                        return Err(ArrowError::CastError(format!(
386                            "Failed to extract primitive of type {} from variant {:?} at path VariantPath([])",
387                            $type_name,
388                            $value
389                        )));
390                    }
391                    // Safe casting: append null on conversion failure
392                    self.builder.append_null();
393                    Ok(false)
394                }
395            }
396
397            fn finish(mut self) -> Result<ArrayRef> {
398                Ok(Arc::new(self.builder.finish()))
399            }
400        }
401    }
402}

434/// Builder for creating VariantArray output (for path extraction without type conversion)
435pub(crate) struct VariantToBinaryVariantArrowRowBuilder {
436    metadata: BinaryViewArray,
437    builder: VariantValueArrayBuilder,
438    nulls: NullBufferBuilder,
439}
440
441impl VariantToBinaryVariantArrowRowBuilder {
442    fn new(metadata: BinaryViewArray, capacity: usize) -> Self {
443        Self {
444            metadata,
445            builder: VariantValueArrayBuilder::new(capacity),
446            nulls: NullBufferBuilder::new(capacity),
447        }
448    }
449}
450
451impl VariantToBinaryVariantArrowRowBuilder {
452    fn append_null(&mut self) -> Result<()> {
453        self.builder.append_null();
454        self.nulls.append_null();
455        Ok(())
456    }
457
458    fn append_value(&mut self, value: Variant<'_, '_>) -> Result<bool> {
459        self.builder.append_value(value);
460        self.nulls.append_non_null();
461        Ok(true)
462    }
463
464    fn finish(mut self) -> Result<ArrayRef> {
465        let variant_array = VariantArray::from_parts(
466            self.metadata,
467            Some(self.builder.build()?),
468            None, // no typed_value column
469            self.nulls.finish(),
470        );
471
472        Ok(ArrayRef::from(variant_array))
473    }
